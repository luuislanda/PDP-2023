  var parsonsPuzzle = new ParsonsWidget({

    "max_wrong_lines": 10,
    "grader": ParsonsWidget._graders.LineBasedGrader,

    "can_indent": true,

      parsonsPuzzle.shuffleLines(); 
  });

      event.preventDefault(); 
      parsonsPuzzle.getFeedback(); 
  });
